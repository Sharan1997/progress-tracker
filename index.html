  <script>

    const allTasks = ["Skin Care", "Workout", "Dry Fruits", "Read 30 min"];


    function formatDate(dateStr) {

      const date = new Date(dateStr);

      const day = date.getDate();

      const suffix = (d => {

        if (d > 3 && d < 21) return 'th';

        switch (d % 10) {

          case 1: return 'st';

          case 2: return 'nd';

          case 3: return 'rd';

          default: return 'th';

        }

      })(day);


      const monthNames = ["Jan", "Feb", "Mar", "Apr", "May", "Jun",

                          "Jul", "Aug", "Sep", "Oct", "Nov", "Dec"];

      const month = monthNames[date.getMonth()];

      const year = date.getFullYear();


      return `${day}${suffix} ${month} ${year}`;

    }


    function saveProgress() {

      const user = document.getElementById('user').value;

      if (!user) {

        alert("Please select a user");

        return;

      }


      const checkboxes = document.querySelectorAll('.tasks input[type="checkbox"]');

      const selectedTasks = Array.from(checkboxes).filter(cb => cb.checked).map(cb => cb.value);

      const today = new Date().toISOString().slice(0, 10);


      const key = `${user}_${today}`;

      const data = {

        user,

        date: today,

        tasks: selectedTasks

      };


      localStorage.setItem(key, JSON.stringify(data));

      document.getElementById('message').innerText = "✅ Today's progress saved!";

      loadEntries();

    }


    function clearCheckboxes() {

      const checkboxes = document.querySelectorAll('.tasks input[type="checkbox"]');

      checkboxes.forEach(cb => cb.checked = false);

      document.getElementById('message').innerText = "☑️ Selections cleared!";

    }


    function loadEntries() {

      const entriesDiv = document.getElementById('entries');

      entriesDiv.innerHTML = '';


      const users = ['Nagaveni', 'Sharan'];

      const dataByDate = {};


      for (let i = 0; i < localStorage.length; i++) {

        const key = localStorage.key(i);

        const [user, date] = key.split('_');

        if (users.includes(user)) {

          const entry = JSON.parse(localStorage.getItem(key));

          if (!dataByDate[date]) dataByDate[date] = {};

          dataByDate[date][user] = entry;

        }

      }


      const sortedDates = Object.keys(dataByDate).sort((a, b) => new Date(b) - new Date(a));


      if (sortedDates.length === 0) {

        entriesDiv.innerHTML = "<p>No entries found.</p>";

        return;

      }


      sortedDates.forEach(date => {

        const dayBlock = document.createElement('div');

        dayBlock.className = 'day-block';


        const formattedDate = formatDate(date);


        const nagaveniData = dataByDate[date]['Nagaveni'] || { tasks: [] };

        const sharanData = dataByDate[date]['Sharan'] || { tasks: [] };


        const renderTasks = (userData) => {

          return allTasks.map(task => {

            const done = userData.tasks.includes(task);

            return `<span class="${done ? 'done' : 'not-done'}">${done ? '✔️' : '❌'} ${task}</span>`;

          }).join("");

        };


        dayBlock.innerHTML = `

          <div class="date-header">${formattedDate}</div>

          <div class="entry-row">

            <div class="entry-column">

              <h4>Nagaveni</h4>

              <div class="task-status">${renderTasks(nagaveniData)}</div>

            </div>

            <div class="entry-column">

              <h4>Sharan</h4>

              <div class="task-status">${renderTasks(sharanData)}</div>

            </div>

          </div>

        `;


        entriesDiv.appendChild(dayBlock);

      });

    }


    window.onload = loadEntries;
